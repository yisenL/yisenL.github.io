# ts常用语法

# keyof 和 in

# keyof：获取 interface 的键

用法：
interface Point {
  x: number;
  y: number;
}

type keys = keyof Point;
// type keys = "x" | "y"

# in：遍历枚举类型

用法：
type Keys = "a" | "b"
type Obj =  {
  [p in Keys]: any
} // -> { a: any, b: any }

# Required & Partial

# Required：将类型中所有选项变为必选，去除所有？

type Required<T> = {
  [P in keyof T]-?: T[P];
};

用法：
interface RequiredContact= Required<Contact>

# Partial：将类型中所有选项变为可选，即加上？

type Partial<T> = { [P in keyof T]?: T[P] };
上面语句的意思是 keyof T 拿到 T 所有属性名，然后 in 进行遍历，将值赋给 P，最后 T[P] 取得相应属性的值

用法：
interface PartialContact= Partial<Contact>

# Pick & Omit

# Pick：选择其中属性

type Pick<T, K extends keyof T> = {
    [P in K]: T[P];
};

用法：
interface ContactPick extends Pick<Contact, 'name' | 'phone'> {}
type ProduceParams = Pick<MemberListParams, 'productName' | 'productSku'>

# Omit：剔除其中属性

type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>

用法：
type OmitEmailContact = Omit<Contact, 'email' | 'phone'>
interface ContactPick extends Omit<Contact, 'name' | 'phone'> {}
